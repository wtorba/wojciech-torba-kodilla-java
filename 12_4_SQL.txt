SELECT I.*, IL.NAME
FROM ISSUES I
INNER JOIN ISSUESLISTS IL
   ON I.ISSUELIST_ID = IL.ID;
   
SELECT I.*, U.FIRSTNAME, U.LASTNAME
FROM ISSUES I
INNER JOIN USERS U
   ON I.USER_ID_ASSIGNEDTO = U.ID;
   
SELECT U.FIRSTNAME, U.LASTNAME, COUNT(*) CNT
FROM USERS U
INNER JOIN ISSUES I
   ON U.ID = I.USER_ID_ASSIGNEDTO
GROUP BY 1, 2;
